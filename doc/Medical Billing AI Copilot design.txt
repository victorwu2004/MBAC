## Phase 2: Claim Verification Engine

### Claim Validation Logic (Rule Layers)

```
Layer 1 - Structural Validation   → EDI 837 schema checks (via Edifecs)
Layer 2 - Code Validity           → ICD-10/CPT exist in current year's table
Layer 3 - Medical Necessity       → Diagnosis ↔ Procedure pairing rules (LCD/NCD)
Layer 4 - Payer Policy Rules      → Payer-specific edits (loaded from config)
Layer 5 - AI Copilot Analysis     → Claude-powered narrative reasoning & suggestion
```

```python
# claim_verifier.py
from dataclasses import dataclass
from typing import Optional
import boto3

@dataclass
class ClaimLine:
    cpt_code: str
    icd10_codes: list[str]
    units: int
    modifier: Optional[str]
    service_date: str

@dataclass
class VerificationResult:
    status: str            # VALID | WARNING | DENIED_LIKELY | ERROR
    errors: list[str]
    warnings: list[str]
    ai_narrative: str      # Claude explanation
    suggested_codes: list[str]

class ClaimVerifier:
    def __init__(self, code_store, ai_client, rules_engine):
        self.codes = code_store        # DynamoDB or RDS lookup
        self.ai = ai_client            # Bedrock / Anthropic
        self.rules = rules_engine

    def verify(self, claim: dict) -> VerificationResult:
        errors, warnings = [], []

        for line in claim["lines"]:
            # Layer 2: Code validity
            if not self.codes.is_valid_cpt(line.cpt_code, line.service_date):
                errors.append(f"CPT {line.cpt_code} invalid for DOS {line.service_date}")

            for dx in line.icd10_codes:
                if not self.codes.is_valid_icd10(dx, line.service_date):
                    errors.append(f"ICD-10 {dx} invalid or not in scope")

            # Layer 3: Medical necessity
            necessity = self.rules.check_necessity(line.cpt_code, line.icd10_codes)
            if not necessity.passes:
                warnings.append(necessity.reason)

        # Layer 5: AI narrative
        narrative = self.ai.analyze_claim(claim, errors, warnings)

        return VerificationResult(
            status="VALID" if not errors else "ERROR",
            errors=errors,
            warnings=warnings,
            ai_narrative=narrative,
            suggested_codes=narrative.get("suggestions", [])
        )
```

### AI Copilot — Bedrock Integration

```python
# ai_copilot.py
import boto3, json

class BillingCopilot:
    def __init__(self, model_id="anthropic.claude-sonnet-4-5"):
        self.bedrock = boto3.client("bedrock-runtime", region_name="us-east-1")
        self.model_id = model_id

    def analyze_claim(self, claim: dict, errors: list, warnings: list) -> dict:
        prompt = self._build_prompt(claim, errors, warnings)
        response = self.bedrock.invoke_model(
            modelId=self.model_id,
            body=json.dumps({
                "anthropic_version": "bedrock-2023-05-31",
                "max_tokens": 1024,
                "messages": [{"role": "user", "content": prompt}]
            })
        )
        return json.loads(response["body"].read())

    def _build_prompt(self, claim, errors, warnings) -> str:
        return f"""
You are a certified medical billing expert. Review this claim:

CLAIM: {json.dumps(claim, indent=2)}
ERRORS: {errors}
WARNINGS: {warnings}

Provide:
1. Plain-language explanation of any issues
2. Suggested ICD-10/CPT code corrections with rationale
3. Likelihood of payer acceptance (High/Medium/Low)
4. Recommended next actions

Respond as JSON: {{"narrative": "", "suggestions": [], "acceptance_likelihood": "", "actions": []}}
"""
```

# infrastructure/cdk/api_stack.py
from aws_cdk import (Stack, aws_lambda as lambda_, aws_apigateway as apigw,
                     aws_dynamodb as ddb, aws_iam as iam)

class CopilotApiStack(Stack):
    def __init__(self, scope, id, **kwargs):
        super().__init__(scope, id, **kwargs)

        # DynamoDB for code lookups
        code_table = ddb.Table(self, "ICD10CodeTable",
            partition_key=ddb.Attribute(name="code", type=ddb.AttributeType.STRING),
            sort_key=ddb.Attribute(name="fiscal_year", type=ddb.AttributeType.STRING),
            billing_mode=ddb.BillingMode.PAY_PER_REQUEST,
        )

        # Lambda function
        fn = lambda_.DockerImageFunction(self, "CopilotFn",
            code=lambda_.DockerImageCode.from_image_asset("../"),
            memory_size=1024,
            timeout=Duration.seconds(30),
        )

        # Grant Bedrock access
        fn.add_to_role_policy(iam.PolicyStatement(
            actions=["bedrock:InvokeModel"],
            resources=["arn:aws:bedrock:us-east-1::foundation-model/anthropic.claude-*"]
        ))
        code_table.grant_read_data(fn)

        # API Gateway
        api = apigw.LambdaRestApi(self, "CopilotApi", handler=fn)

## Phase 5: HIPAA Compliance Checkpoints

- **Bedrock** — HIPAA eligible service; execute a BAA with AWS
- **All S3 buckets** — Server-side encryption (SSE-KMS), no public access
- **RDS Aurora** — Encrypted at rest, in a private subnet, no public endpoint
- **API Gateway** — TLS 1.2+ only, WAF rules blocking OWASP Top 10
- **CloudTrail** — All API calls logged for audit
- **Secrets Manager** — Rotate credentials; never hardcode PHI or keys
- **VPC Flow Logs** — Network-level audit trail
